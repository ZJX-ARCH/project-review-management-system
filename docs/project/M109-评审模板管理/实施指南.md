  // æ€»åˆ†è®¡ç®—
  const calculatedTotalScore = computed(() => {

  // æ·»åŠ å¤§ç±»

      items: []
    })
  }

  // æ ¡éªŒè¡¨å•

      return false
    }
    return true
  }

  return {

    categories,
    calculatedTotalScore,
    totalScoreClass,

  }
